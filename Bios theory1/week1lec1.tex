    $cov (X,Y)=E((x-\mu_{x})(Y-\mu_{y}))$, $corr (X,Y)=\frac{cov (X,Y)}{\sigma_{x}\sigma_{y}} $
\end{definition}

Basic Fact:
\begin{enumerate}
    \item $cov (X,Y)=E(XY)-E(X)E(Y) $
    \item $cov (X,Y)=0 \text{ if X,Y is i.d.}$
    
\end{enumerate}
\begin{proof}
    \begin{align*}
        cov (X,Y)&= E((X-\mu_{x}) (Y-\mu_{Y})) \\

        &= E( XY )  -\mu_{x}\mu_{y} \\
    \end{align*}
\end{proof}
